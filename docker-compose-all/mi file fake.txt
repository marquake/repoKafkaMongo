version: '3.8'

services:
  kafka:
    image: wurstmeister/kafka
    ports:
      - "9092:9092"
    environment:
      KAFKA_ADVERTISED_HOST_NAME: localhost
      KAFKA_AUTO_CREATE_TOPICS_ENABLE: "true"
      KAFKA_ZOOKEEPER_CONNECT: zookeeper:2181

  zookeeper:
    image: wurstmeister/zookeeper
    ports:
      - "2181:2181"

  mongo:
    image: mongo
    ports:
      - "27017:27017"

  cliente:
    build:
      context: ./proyectoCliente
      dockerfile: DockerfileB
    depends_on:
      - kafka
      - mongo

  repartidor:
    build:
      context: ./proyectoRepartidor
      dockerfile: DockerfileC
    depends_on:
      - kafka
      - mongo

  pedido:
    build:
      context: ./proyectoPedido
      dockerfile: DockerfileC
    depends_on:
      - kafka
      - mongo
